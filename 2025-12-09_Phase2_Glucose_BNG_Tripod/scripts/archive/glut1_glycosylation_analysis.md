import mdtraj as md

def analyze_blocking_effect(traj, ligand_name='LIG'):
    """
    Glycanì´ Tripod ì§„ì…ì„ ì°¨ë‹¨í•˜ëŠ”ì§€ ë¶„ì„
    """

    )

        'resname ANE5AC and resid 288'

        'resid 168 169 288 289 290'
